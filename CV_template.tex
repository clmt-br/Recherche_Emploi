\newcommand{\expentry}[5]{%
  \noindent\begin{minipage}[t]{\dimexpr\textwidth-\datecol-4pt\relax}%
    \textbf{#1} \textit{- #2}\\[-5pt]
    {\small\textcolor{lightgray}{#3}}%
    \begin{itemize}[leftmargin=1.1em, itemsep=1.5pt, topsep=4pt, parsep=0pt, partopsep=0pt, label=\textbullet]
      #5
    \end{itemize}%
  \end{minipage}%
  \hfill%
  \begin{minipage}[t]{\datecol}%
    \raggedleft\small\textcolor{lightgray}{\mbox{#4}}%
  \end{minipage}%
  \par
}

% Version sans bullets

}
